def crossover(parent1, parent2):
    """
    –°–æ–∑–¥–∞–Ω–∏–µ –ø–æ—Ç–æ–º–∫–∞ –∏–∑ –¥–≤—É—Ö —Ä–æ–¥–∏—Ç–µ–ª–µ–π

            child[indicator] = parent2[indicator].copy()
    
    # –ú—É—Ç–∞—Ü–∏—è –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –≤–Ω—É—Ç—Ä–∏ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–∞

        # –°–º–µ—à–∏–≤–∞–Ω–∏–µ –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤
        for param in child[indicator]:
            if random.random() < 0.5:
                child[indicator][param] = parent1[indicator][param]
            else:
                child[indicator][param] = parent2[indicator][param]
    
    return child

def calculate_pnl(positions_history, initial_capital=10000):
    """
    –†–∞—Å—á–µ—Ç –ø—Ä–∏–±—ã–ª—å–Ω–æ—Å—Ç–∏ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏
